/* Utilitários de data */
function dateKey(d){
    const y=d.getFullYear(), m=String(d.getMonth()+1).padStart(2,'0'), dy=String(d.getDate()).padStart(2,'0');
    return `${y}-${m}-${dy}`;
}
function isToday(d){ return dateKey(d)===dateKey(new Date()); }
function fmtLabel(d){
    if(isToday(d)) return '📅 Hoje';
    return d.toLocaleDateString('pt-BR',{weekday:'short',day:'2-digit',month:'2-digit',year:'2-digit'});
}

/* Init */
function init(){
    const g = localStorage.getItem('nt_g');
    if(g) goals = JSON.parse(g);
    goalCal.value   = goals.cal;
    goalProt.value  = goals.prot;
    goalWater.value = goals.water || 8;

    const mf = localStorage.getItem('nt_myfoods');
    myFoods = mf ? JSON.parse(mf) : [];
    myFoods.sort((a,b)=>a.n.localeCompare(b.n,'pt'));

    const pw = localStorage.getItem('nt_peso');
    if(pw) pesoMetaData = JSON.parse(pw);
    if(pesoMetaData.pesoAtual) pesoAtual.value = pesoMetaData.pesoAtual;
    if(pesoMetaData.pesoMeta)  pesoMeta.value  = pesoMetaData.pesoMeta;
    if(pesoMetaData.fatorProt) fatorProt.value = pesoMetaData.fatorProt;

    buildSelect();
    buildQuickList();
    document.querySelectorAll('.meal-tab').forEach(btn=>{
        btn.onclick = ()=>{
            document.querySelectorAll('.meal-tab').forEach(x=>x.classList.remove('on'));
            btn.classList.add('on');
            meal = btn.dataset.meal;
            foodSel.value = '';
        };
    });

    updateBankCount();
    loadViewDay();
    upd();
    renderWeekSummary();
}

/* Dia atual na view */
function loadViewDay(){
    const key = dateKey(viewDate);
    const f = localStorage.getItem('nt_'+key);
    foods = f ? JSON.parse(f) : [];
    const w = localStorage.getItem('nt_w_'+key);
    water = w ? parseInt(w) : 0;

    navLabel.textContent = fmtLabel(viewDate);
    dayTitle.textContent = isToday(viewDate) ? 'Hoje' :
        viewDate.toLocaleDateString('pt-BR',{weekday:'long',day:'2-digit',month:'2-digit'});
    histNote.textContent = isToday(viewDate) ? '' : '👁 Histórico — somente leitura';

    addCard.style.opacity = isToday(viewDate) ? '1' : '0.4';
    addCard.style.pointerEvents = isToday(viewDate) ? 'auto' : 'none';

    render();
    renderWater();
}

function saveDay(){
    const key = dateKey(viewDate);
    localStorage.setItem('nt_'+key, JSON.stringify(foods));
    localStorage.setItem('nt_w_'+key, water);
}
